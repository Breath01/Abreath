实现strStr()

给定一个 haystack 字符串和一个 needle 字符串，在 haystack 字符串中找出 needle 字符串出现的第一个位置 (从0开始)。如果不存在，则返回  -1。 著作权归领扣网络所有。商业转载请联系官方授权，非商业转载请注明出处。

样例

输入: haystack = "hello", needle = "ll"
输出: 2

C++思路一

class Solution {
public:
    int strStr(string haystack, string needle) {
        if(needle.size()==0) return 0;
        if(needle.size()>haystack.size()) return -1;
        int ans=-1;
        int f=1;
        for(int i=0;i<haystack.size()-needle.size()+1;++i){
            f=1;
            for(int j=0;j<needle.size();++j){
                if(haystack[i+j]!=needle[j]){
                    f=0;
                    continue;
                }
            }
            if(f==1){
                ans=i;
                break;
            } 
        }
        return ans;
    }
};

C++思路二

class Solution {
public:
    int next[100005];
    void get_next(string n){
        int i=0,j=-1;
        next[i]=-1;
        while(i<size(n)){
            if(j==-1||n[i]==n[j]){
                ++i;++j;
                next[i]=j;
            }else{
                j=next[j];
            }
        }
    }
    int strStr(string haystack, string needle) {
        get_next(needle);
        int i=0,j=0;
        int sz1=size(haystack),sz2=size(needle);
        while(i<sz1&&j<sz2){
            if(j==-1||haystack[i]==needle[j]){
                ++i;++j;
            }else{
                j=next[j];
            }
        }
        if(j+1>sz2) return i-sz2;
        else return -1;
    }
};
