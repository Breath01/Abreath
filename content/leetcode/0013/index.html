罗马数字转整数

输出: 3

解题思路

C++思路一

特判4，9，40，90，400，900。时间复杂度O(len(N))，超过82.31%，空间复杂度O(1)，超过84.41%

代码

C++思路一

class Solution {
public:
    int romanToInt(string s) {
        int ans=0;
        int cur=0;
        while(cur<size(s)){
            if(s[cur]=='C'&&s[cur+1]=='M'){
                ans+=900;
                cur+=2;
                continue;
            }else if(s[cur]=='C'&&s[cur+1]=='D'){
                ans+=400;
                cur+=2;
                continue;
            }else{

            }
            if(s[cur]=='X'&&s[cur+1]=='C'){
                ans+=90;
                cur+=2;
                continue;
            }else if(s[cur]=='X'&&s[cur+1]=='L'){
                ans+=40;
                cur+=2;
                continue;
            }else{

            }
            if(s[cur]=='I'&&s[cur+1]=='X'){
                ans+=9;
                cur+=2;
                continue;
            }else if(s[cur]=='I'&&s[cur+1]=='V'){
                ans+=4;
                cur+=2;
                continue;
            }else{

            }
            if(s[cur]=='M'){
                ans+=1000;
                cur+=1;
                continue;
            }
            if(s[cur]=='D'){
                ans+=500;
                cur+=1;
                continue;
            }
            if(s[cur]=='C'){
                ans+=100;
                cur+=1;
                continue;
            }
            if(s[cur]=='L'){
                ans+=50;
                cur+=1;
                continue;
            }
            if(s[cur]=='X'){
                ans+=10;
                cur+=1;
                continue;
            }
            if(s[cur]=='V'){
                ans+=5;
                cur+=1;
                continue;
            }
            if(s[cur]=='I'){
                ans+=1;
                cur+=1;
                continue;
            }
        }
        return ans;
    }
};
